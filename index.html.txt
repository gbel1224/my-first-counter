    <div class="counter-container">
        <h1>Counter</h1>
        <div id="counter-value">0</div>
        
        <button class="btn-decrease" onclick="changeCount(-1)">- Decrease</button>
        <button class="btn-increase" onclick="changeCount(1)">+ Increase</button>
    </div>

    <script>
        // JavaScript: The logic
        let count = 0; // Our starting number
        const display = document.getElementById('counter-value');

        function changeCount(amount) {
            // 1. Update the math
            count = count + amount;
            
            // 2. Update the screen
            display.textContent = count;
            
            // 3. Optional: Change color if negative
            if (count < 0) {
                display.style.color = 'red';
            } else {
                display.style.color = '#333';
            }
        }
    </script>
